Q-11 What are all the looping structures in JavaScript?

Types Of Loop
1:-For Loop:-
=>Syntax:-
for (initialization; condition; increment/decrement)
{
//code
}
=>>Example:-
for (let i = 0; i < 5; i++)
{
console.log(i);
}

2:-while Loop:-
=>Syntax:-
while(condition) {
//code
}
=>>Example:-
let i = 0;
while(i < 5)
{
console.log(i);
i++;
}

3:-Do While Loop:-
=>Syntax:-
do {
//code
} while(condition)
=>>Example:-
let i = 0;
do {
console.log(i);
i++;
} while(i < 5)

4:-For in:-
=>Syntax:-
for(variable in object)
{
//code
}
=>>Example:-
let person = {name:"John", age:30, city:"New York" };
for(let key in person)
{
console.log(`${key}: ${person[key]}`);
}

5:-For of:-
=>Syntax:-
for(variable of iterable)
{
//code
}
=>>Example:-
let arr = [10, 20, 30, 40, 50];
for(let value of arr)
{
console.log(value);
}
